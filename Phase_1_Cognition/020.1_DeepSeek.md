def enriched_model(self, beta, D, log_tau0, c, sigma=0.5):

"""
u = np.log10(beta)

def compute_curvature(self, smooth=True):

def bic_comparison(self):
"""
Comparaison BIC (parcimonie)
"""
if not hasattr(self, 'L_enr'):
self.fit_enriched()

n = self.n

Baseline : 1 paramètre (τ0)

rss_base = np.sum(resid_base**2)
self.bic_base = n * np.log(rss_base / n) + 1 * np.log(n)

Enrichi : 1 paramètre baseline + 1 paramètre c = 2 paramètres

rss_enr = np.sum(resid_enr**2)
self.bic_enr = n * np.log(rss_enr / n) + 2 * np.log(n)

self.bic_gain = self.bic_base - self.bic_enr
return self.bic_gain
